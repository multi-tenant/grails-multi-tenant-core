001    package grails.plugin.multitenant.core.spring;
002    
003    import org.aopalliance.intercept.MethodInterceptor;
004    import org.aopalliance.intercept.MethodInvocation;
005    
006    import java.lang.reflect.Method;
007    
008    /**
009     * Method interceptor used with Spring AOP to proxy beans that need to be tenant-aware
010     */
011    public class TenantMethodInterceptor implements MethodInterceptor {
012    // ========================================================================================================================
013    //    Instance Fields
014    // ========================================================================================================================
015    
016        /**
017         * The old bean name
018         */
019        String oldBeanName;
020    
021        /**
022         * The container that contains the target bean per tenant.
023         */
024        TenantBeanContainer tenantBeanContainer;
025    
026    // ========================================================================================================================
027    //    Public Instance Methods
028    // ========================================================================================================================
029    
030        public Object invoke(MethodInvocation methodInvocation) throws Throwable {
031            //Get the appropriate bean for the current tenant
032            final Object beanFromContainer = tenantBeanContainer.getBean(oldBeanName);
033    
034            //Identify the corresponding method we are trying to call
035            final Method method = beanFromContainer.getClass().getMethod(
036                    methodInvocation.getMethod().getName(),
037                    methodInvocation.getMethod().getParameterTypes());
038    
039            //Invoke the method and return the value
040            return method.invoke(beanFromContainer, methodInvocation.getArguments());
041        }
042    
043        public void setOldBeanName(String oldBeanName) {
044            this.oldBeanName = oldBeanName;
045        }
046    
047        public void setTenantBeanContainer(TenantBeanContainer tenantBeanContainer) {
048            this.tenantBeanContainer = tenantBeanContainer;
049        }
050    }
