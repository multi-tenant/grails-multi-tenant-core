001    package grails.plugin.multitenant.core.util.log.log;
002    
003    import grails.plugin.multitenant.core.util.TenantUtils;
004    import org.apache.log4j.helpers.FormattingInfo;
005    import org.apache.log4j.helpers.PatternConverter;
006    import org.apache.log4j.helpers.PatternParser;
007    import org.apache.log4j.spi.LoggingEvent;
008    
009    /**
010     * Class for parsing patterns from log formats and injecting the current tenant's id.  The pattern for
011     * displaying the current tenantId is %T
012     */
013    public class MultiTenantPatternParser extends PatternParser
014    {
015    
016        public static final char TENANT_CONVERTER_CHAR = 'T';
017    
018        public MultiTenantPatternParser(String pattern)
019        {
020            super(pattern);
021        }
022    
023        public void finalizeConverter(char c)
024        {
025            if (c == TENANT_CONVERTER_CHAR)
026            {
027                addConverter(new MultiTenantPatternConverter(formattingInfo));
028                currentLiteral.setLength(0);
029            } else
030            {
031                super.finalizeConverter(c);
032            }
033        }
034    
035        /**
036         * This class will actually return the replacement string for the logger (just the integer of the current tenant)
037         */
038        private class MultiTenantPatternConverter extends PatternConverter
039        {
040    
041            MultiTenantPatternConverter(FormattingInfo formattingInfo)
042            {
043                super(formattingInfo);
044            }
045    
046            public String convert(LoggingEvent event)
047            {
048                Integer rtn = TenantUtils.getCurrentTenant();
049                if (rtn == null)
050                {
051                    return "N/A";
052                } else
053                {
054                    return rtn.toString();
055                }
056            }
057        }
058    }
